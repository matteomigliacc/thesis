% AI logbook: rotated running header + page number on landscape pages.
% Defined here (preamble = global scope) so the eso-pic shipout hook can still
% find \iflscapehdr after the appendices group closes. Used by AI_logbook_table.tex.
\newif\iflscapehdr\lscapehdrfalse
\fancypagestyle{lscapeblank}{\fancyhf{}\renewcommand{\headrulewidth}{0pt}\renewcommand{\footrulewidth}{0pt}}
\AddToShipoutPictureFG{%
  \iflscapehdr
    % running header along the landscape top edge (physical left margin)
    \put(\LenToUnit{0.55in},\LenToUnit{0.8in}){\rotatebox{90}{\parbox{\dimexpr\paperheight-1.6in\relax}{\footnotesize\color{headergray}\leftmark}}}%
    % page number centred along the landscape bottom edge (physical right margin)
    \put(\LenToUnit{\dimexpr\paperwidth-0.7in\relax},\LenToUnit{0.8in}){\rotatebox{90}{\makebox[\dimexpr\paperheight-1.6in\relax][c]{\footnotesize\thepage}}}%
  \fi}
\newenvironment{lscapepage}{\clearpage\lscapehdrtrue\begin{landscape}\pagestyle{lscapeblank}}%
  {\end{landscape}\lscapehdrfalse\clearpage\pagestyle{fancy}}
